```tsx
'use client';

import { GanttTask, GanttConfig } from '@/types/gantt';
import { GanttTaskBar } from './gantt-task-bar';
import { Tooltip, TooltipContent, TooltipTrigger } from '@/components/ui/tooltip';
import { cn } from '@/lib/utils';
import { format } from 'date-fns';

/**
 * GanttTaskRow renders a single row in the Gantt chart
 * with task details sidebar and task bar in the timeline.
 * 
 * @param task - The task to render
 * @param config - Gantt configuration for positioning
 * @param onDragStart - Handler for drag start
 * @param onClick - Handler for task click
 */
interface GanttTaskRowProps {
  task: GanttTask;
  config: GanttConfig;
  onDragStart?: (e: React.MouseEvent, task: GanttTask) => void;
  onClick?: (task: GanttTask) => void;
  isBeingDragged?: boolean;
}

export function GanttTaskRow({
  task,
  config,
  onDragStart,
  onClick,
  isBeingDragged,
}: GanttTaskRowProps) {
  return (
    <div
      className={cn(
        "flex items-center border-b border-border",
        isBeingDragged && "opacity-70"
      )}
      style={{ height: config.rowHeight }}
    >
      {/* Task sidebar (fixed width) */}
      <div className="w-[200px] flex-shrink-0 px-3 border-r border-border bg-background">
        <div className="truncate text-sm font-medium">{task.name}</div>
        <div className="text-xs text-muted-foreground">
          {task.assignee?.name || 'Unassigned'}
        </div>
      </div>
      
      {/* Task bar in timeline */}
      <div className="flex-1 relative">
        <Tooltip>
          <TooltipTrigger asChild>
            <GanttTaskBar
              task={task}
              config={config}
              onDragStart={onDragStart}
              onClick={onClick}
            />
          </TooltipTrigger>
          <TooltipContent side="top" className="max-w-[300px]">
            <div className="space-y-1">
              <p className="font-semibold">{task.name}</p>
              <p className="text-xs">
                {format(task.startDate, 'MMM d')} – {format(task.endDate, 'MMM d, yyyy')}
              </p>
              {task.assignee && (
                <p className="text-xs">Assigned to: {task.assignee.name}</p>
              )}
              {task.progress !== undefined && (
                <p className="text-xs">Progress: {task.progress}%</p>
              )}
            </div>
          </TooltipContent>
        </Tooltip>
      </div>
    </div>
  );
}
```
